# Настройка ufw
$ sudo apt install ufw && sudo ufw status verbose
# Какие на этот момент есть правила
$ sudo ufw app list
# Разрешим правило для ssh сервера
$ sudo ufw allow OpenSSH
# Разрешим правило для Samba
$ sudo ufw allow Samba
# Средненькое логирование
$ sudo ufw logging medium
# Ufw на запуск
$ sudo ufw enable
# Перегрузим и посмотрим, что получилось
$ sudo ufw reload && sudo ufw status verbose
