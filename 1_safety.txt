# Настройка ufw
sudo apt install ufw && sudo ufw status
sudo ufw app list
sudo ufw allow OpenSSH
sudo ufw allow Samba
sudo ufw logging medium
sudo ufw enable
sudo ufw reload && sudo ufw status verbose
