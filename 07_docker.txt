# docker
# https://docs.docker.com/desktop/install/ubuntu/
# https://docs.docker.com/engine/install/ubuntu/#set-up-the-repository
sudo apt update && sudo apt upgrade -y && sudo apt install ca-certificates curl gnupg lsb-release
sudo mkdir -p /etc/apt/keyrings && curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg
echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
sudo apt update && sudo apt upgrade -y

# Отличие Docker-Desktop заключается в том, что Desktop перехватывает управление и все свои дела ведет на территории гыука
# Вся работа идентична и там и там. Может возникнуть путаница где какие образы и контейнеры крутятся.
# Docker-Desktop не видит системные, а система не видит Docker-Desktop контейнеры.
# То есть при запущеном Docker-Desktop Вы управляете своими контейнерами, а если нет (или при запушеном Docker-Desktop выполните команду от sudo) - то системными.

# Далее для Docker-Desktop устанавливаем
# https://desktop.docker.com/linux/main/amd64/docker-desktop-4.15.0-amd64.deb?utm_source=docker&utm_medium=webreferral&utm_campaign=docs-driven-download-linux-amd64
ls
sudo apt-get install ./docker-desktop-<version>-<arch>.deb
# Запуск или ярлык Docker-Desktop или systemctl --user start docker-desktop
# Локация $HOME/.docker/

# Для системы ставим
sudo apt-get install docker-ce docker-ce-cli containerd.io docker-compose-plugin
sudo usermod -aG docker $(whoami)
sudo reboot

# Для управления есть web-ка (которую можно юзать и при Docker-Desktop,запущеном без морды: systemctl --user start docker-desktop):
# https://docs.portainer.io/start/install/server/docker/linux
docker volume create portainer_data
docker pull portainer/portainer-ce:latest
docker run -d -p 8000:8000 -p 9443:9443 --name portainer --restart=always -v /var/run/docker.sock:/var/run/docker.sock -v portainer_data:/data portainer/portainer-ce:latest
# Управляется по вебке
http://192.168.0.8:9443

sudo docker info
sudo systemctl stop docker && sudo systemctl status docker
sudo nano /etc/docker/daemon.json
{
    "data-root": "/srv/docker"
}
